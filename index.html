2. Phone Number OTP Login

Enable phone number authentication in Firebase console.

html
<script>
  let confirmationResult;

  function sendOTP() {
    const phoneNumber = document.getElementById('phoneNumber').value;
    const appVerifier = new firebase.auth.RecaptchaVerifier('loginSection', {
      size: 'invisible'
    });
    auth.signInWithPhoneNumber(phoneNumber, appVerifier)
      .then(result => {
        confirmationResult = result;
        document.getElementById('otp').style.display = 'inline-block';
        document.getElementById('verifyBtn').style.display = 'inline-block';
      })
      .catch(error => alert(error.message));
  }

  function verifyOTP() {
    const code = document.getElementById('otp').value;
    confirmationResult.confirm(code).then(result => {
      const user = result.user;
      alert('Login Successful');
      loadUserBookings(user.phoneNumber);
    }).catch(error => alert(error.message));
  }
</script>

3. Booking Handler

Save user booking to Firestore:

html
<script>
  function handleSubmit(e) {
    e.preventDefault();
    const booking = {
      name: document.getElementById('name').value,
      phone: document.getElementById('phone').value,
      address: document.getElementById('address').value,
      service: document.getElementById('service').value,
      date: document.getElementById('date').value,
      time: document.getElementById('time').value,
      notes: document.getElementById('notes').value,
      timestamp: new Date()
    };

    db.collection("bookings").add(booking)
      .then(() => {
        alert("Booking Successful!");
        document.querySelector("form").reset();
      })
      .catch(error => alert("Error: " + error.message));
  }
</script>

4. Load Bookings in Dashboard

html
<script>
  function loadUserBookings(phoneNumber) {
    db.collection("bookings")
      .where("phone", "==", phoneNumber)
      .get()
      .then(snapshot => {
        const container = document.getElementById('ordersList');
        container.innerHTML = '';
        snapshot.forEach(doc => {
          const data = doc.data();
          container.innerHTML += `
            <div class="order-item">
              <strong>${data.service}</strong> on ${data.date} at ${data.time}
              <div class="order-actions">
                <button class="cancel-btn" onclick="cancelBooking('${doc.id}')">Cancel</button>
                <button class="reschedule-btn" onclick="rescheduleBooking('${doc.id}')">Reschedule</button>
              </div>
            </div>`;
        });
      });
  }

  function cancelBooking(id) {
    db.collection("bookings").doc(id).delete().then(() => {
      alert("Booking cancelled");
      location.reload();
    });
  }

  function rescheduleBooking(id) {
    const newDate = prompt("Enter new date (YYYY-MM-DD):");
    const newTime = prompt("Enter new time (HH:MM):");
    if (newDate && newTime) {
      db.collection("bookings").doc(id).update({
        date: newDate,
        time: newTime
      }).then(() => {
        alert("Rescheduled");
        location.reload();
      });
    }
  }
</script>

html

        });
      });
  }

  // Optional: Auto-load for admin on page load
  window.onload = function() {

  };
</script>
